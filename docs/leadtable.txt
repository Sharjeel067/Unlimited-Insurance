create table public.leads (
  id uuid not null default gen_random_uuid (),
  submission_id text null,
  submission_date timestamp with time zone null,
  customer_full_name text null,
  street_address text null,
  city text null,
  state text null,
  zip_code text null,
  phone_number text null,
  email text null,
  date_of_birth date null,
  age integer null,
  social_security text null,
  health_conditions text null,
  carrier text null,
  product_type text null,
  coverage_amount numeric null,
  monthly_premium numeric null,
  draft_date text null,
  future_draft_date text null,
  beneficiary_routing text null,
  beneficiary_account text null,
  additional_notes text null,
  center text null,
  buffer_agent text null,
  agent text null,
  user_id text null,
  created_at timestamp with time zone null default now(),
  updated_at timestamp with time zone null default now(),
  birth_state text null,
  driver_license text null,
  existing_coverage text null,
  previous_applications text null,
  height text null,
  weight text null,
  doctors_name text null,
  tobacco_use text null,
  medications text null,
  beneficiary_information text null,
  institution_name text null,
  account_type text null,
  is_callback boolean null default false,
  is_retention_call boolean null default false,
  constraint leads_pkey primary key (id),
  constraint leads_submission_id_key unique (submission_id)
) TABLESPACE pg_default;
