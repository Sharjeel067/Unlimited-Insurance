create table public.call_results (
  id uuid not null default gen_random_uuid (),
  lead_id uuid null,
  submission_id text null,
  application_submitted boolean null default false,
  status character varying null,
  notes text null,
  carrier character varying null,
  product_type character varying null,
  draft_date character varying null,
  submitting_agent character varying null,
  licensed_agent_account character varying null,
  coverage_amount numeric null,
  monthly_premium numeric null,
  face_amount numeric null,
  submission_date timestamp with time zone null,
  agent_id character varying null,
  user_id uuid null,
  created_at timestamp with time zone null default now(),
  updated_at timestamp with time zone null default now(),
  buffer_agent text null,
  agent_who_took_call text null,
  sent_to_underwriting boolean null default false,
  call_source text null,
  dq_reason text null,
  new_draft_date date null,
  is_callback boolean null default false,
  is_retention_call boolean null default false,
  carrier_attempted_1 character varying(255) null,
  carrier_attempted_2 character varying(255) null,
  carrier_attempted_3 character varying(255) null,
  constraint call_results_pkey primary key (id),
  constraint call_results_lead_id_fkey foreign KEY (lead_id) references leads (id),
  constraint call_results_user_id_fkey foreign KEY (user_id) references auth.users (id)
) TABLESPACE pg_default;

create index IF not exists idx_call_results_lead_id on public.call_results using btree (lead_id) TABLESPACE pg_default;

create index IF not exists idx_call_results_submission_id on public.call_results using btree (submission_id) TABLESPACE pg_default;

create index IF not exists idx_call_results_user_id on public.call_results using btree (user_id) TABLESPACE pg_default;
